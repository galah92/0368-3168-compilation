class Tree {
Tree left  := nil;
Tree right := nil;
int weight := 0;
}

int sum(Tree t) {
	Tree left := t.left;
	Tree right := t.right;
	int weight := t.weight;
	
	int res := weight;
	if (1 - (left = nil))
	{
		res := res + sum(left);
	}
	if (1 - (right = nil))
	{
		res := res + sum(right);
	}
	return res; }

array TreeArray = Tree[]

void main()
{
	Tree root := new Tree;
	TreeArray TA := new Tree[17];

	int i := 0;

	while (i < 17)
	{
		TA[i] := new Tree;
		TA[i].weight := 2*i;
		i := i+1;
	}

	root.left  := TA[0];
	root.right := TA[1];

	PrintInt(sum(root));
}
